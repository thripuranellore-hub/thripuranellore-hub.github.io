    <script>
        fetch("https://dummyjson.com/products?limit=50&&skip=50")
            .then(function (res) { return res.json() })
            .then(function (data) {
                genUI(data.products);
            })
        console.log("Hello");
        function genUI(products) {
            products.forEach(function (product) {
                var nli = document.createElement('li')
                nli.innerText = product.title
                document.getElementById("d1").appendChild(nli)
            })
        }
    </script>
